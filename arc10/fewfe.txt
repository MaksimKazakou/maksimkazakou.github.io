    <script>
        function sprawdz() {
            var rok = document.getElementById("rok").value;
            var imie = document.getElementById("imie").value;
            var wynik = document.getElementById("wynik");

            const rokUrodzeniaPele = 1940;
            const smiercRok = 2022;
            var wiek = rok - rokUrodzeniaPele;
            
            if (rok <= smiercRok) {
                wynik.innerHTML = " Hej, " + imie + ", Pele miał " + wiek + " lat w " + rok + " roku.";
            } else {
                wynik.innerHTML = " Hej, " + imie + ", Pele miałby " + wiek + " lat w " + rok + " roku.";
            }
        }
    </script>

    <input type="button" onclick="sprawdz()" value="sprawdz">
